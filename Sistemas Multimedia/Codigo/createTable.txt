create table comics(id integer not null,
		    nombre varchar(255) primary key,
		    fecha datetime not null,
		    comic_small varchar(255),
		    comic_medium varchar(255),
		    comic_big varchar(255),
		    tags varchar(255),
		    descripcion text);
create table fanarts(id integer not null,
		    nombre varchar(255) primary key,
		    fecha datetime not null,
		    fanart_small varchar(255),
		    fanart_big varchar(255),
		    tags varchar(255),
		    descripcion text);
create table comentarios(id integer not null auto_increment,
			comic varchar(255) not null,
            autor varchar(255),
            fecha datetime not null,
            texto text,
            email varchar(255),
			primary key (id),
			foreign key (comic) references comics(nombre));
create table comentarios_fanart(id integer not null auto_increment,
			fanart varchar(255) not null,
            autor varchar(255),
            fecha datetime not null,
            texto text,
            email varchar(255),
			primary key (id),
			foreign key (fanart) references fanarts(nombre));
create table administracion(id integer not null auto_increment,
			texto_autor1 text,
			texto_autor2 text,
			imagen_autor1 varchar(255),
			imagen_autor2 varchar(255),
			facebook varchar(255),
			twitter varchar(255),
			deviantart varchar(255),
			imagen_portada varchar(255),
			imagen_mascara varchar(255),
			user varchar(255),
			pass varchar(255),
			primary key (id));
create table noticias(id integer not null,
            titulo varchar(255),
			corto text,
			texto text,
			fecha datetime not null,
			primary key (titulo));
create table comentarios_noticia(id integer not null auto_increment,
			noticia varchar(255) not null,
            autor varchar(255),
            fecha datetime not null,
            texto text,
            email varchar(255),
			primary key (id),
			foreign key (noticia) references noticias(titulo));
